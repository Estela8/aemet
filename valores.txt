 CREATE TABLE IF NOT EXISTS valores (
        valores_id SMALLINT UNSIGNED AUTO_INCREMENT,
        fecha VARCHAR (50),
        indicativo VARCHAR (50),
        nombre VARCHAR (50), 
        provincia VARCHAR (50),
        altitud INT, 
        tmed FLOAT, 
        prec FLOAT,
        tmin FLOAT,
        horatmin VARCHAR (50),
        tmax FLOAT,
        horatmax VARCHAR (50),
        dir VARCHAR (50),
        velmedia FLOAT,
        racha FLOAT,
        horaracha VARCHAR (50),
        hrMedia FLOAT,
        hrMax FLOAT,
        horaHrMax VARCHAR (50),
        hrMin FLOAT,
        horaHrMin VARCHAR (50),
        PRIMARY KEY (valores_id) 
        );
