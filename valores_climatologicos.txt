CREATE TABLE IF NOT EXISTS valores_climatologicos (
    valores_climatologicos_id SMALLINT UNSIGNED AUTO_INCREMENT,
    fecha VARCHAR(50),
    altitud INT,
    tmed VARCHAR(50),
    prec VARCHAR(50),
    tmin VARCHAR(50),
    horatmin VARCHAR(50),
    tmax VARCHAR(50),
    horatmax VARCHAR(50),
    dir FLOAT,
    velemedia VARCHAR(50),
    racha VARCHAR(50),
    horaracha VARCHAR(50),
    hrMedia FLOAT,
    hrMax FLOAT,
    horaHrMax VARCHAR(50),
    hrMin FLOAT,
    horaHrMin VARCHAR(50),
    indicativo_id INT,
    nombre_id INT,
    provincia_id INT,
    PRIMARY KEY (valores_climatologicos_id),
    FOREIGN KEY (indicativo_id) REFERENCES indicativo(indicativo_id),
    FOREIGN KEY (nombre_id) REFERENCES nombre(nombre_id),
    FOREIGN KEY (provincia_id) REFERENCES provincia(provincia_id)
);
